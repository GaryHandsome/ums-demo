1、创建数据库及相关的数据表
    create database ums ;

    use ums ;

    create table userinfo
    (
        id int identity(1,1) primary key ,
        username varchar(50) ,
        password char(6),
        user_age int ,
        user_sex char(2) default '男',
        weight float
    ) ;

    insert into userinfo(username,password,user_age,user_sex,weight)
        values ('admin','123456',18,'男',60.5) ;
